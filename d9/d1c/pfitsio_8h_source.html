203  Writes an image to a fits file.
204  image:: image data, a 2d Image.
205  header:: structure containing header information
206  overwrite:: if true, overwrites old fits file with same name
207 */
208  if (image.size() != rows * cols * chans)
209  throw std::runtime_error("image or cube size does not match dimensions.");
210 
211  PURIFY_LOW_LOG("Writing fits file {}", header.fits_name);
212  if (overwrite == true) {
213  remove(header.fits_name.c_str());
214  }
215  fitsfile *fptr;
216  t_int status = 0;
217  std::vector<long> naxes = {static_cast<long>(rows), static_cast<long>(cols),
218  static_cast<long>(chans), 1};

220 
221  if (fits_create_file(&fptr, header.fits_name.c_str(), &status))
222  throw std::runtime_error("Problem creating fits file:" + header.fits_name);
223  if (fits_create_img(fptr, DOUBLE_IMG, static_cast<t_int>(naxes.size()), naxes.data(), &status))
224  throw std::runtime_error("Problem creating HDU for image in fits file:" + header.fits_name);
225  if (fits_write_pix(fptr, TDOUBLE, fpixel.data(), static_cast<long>(image.size()),
226  const_cast<t_real *>(image.derived().data()), &status))
227  throw std::runtime_error("Problem writing image in fits file:" + header.fits_name);
228 
229 #define PURIFY_MACRO(KEY, VALUE, COMMENT) write_key(fptr, KEY, VALUE, COMMENT, &status);
230

233  PURIFY_MACRO("NAXIS", static_cast<t_int>(naxes.size()), ""); // number of axes
234  PURIFY_MACRO("NAXIS1", static_cast<t_int>(naxes.at(0)), "");
235  PURIFY_MACRO("NAXIS2", static_cast<t_int>(naxes.at(1)), "");
236  PURIFY_MACRO("NAXIS3", static_cast<t_int>(naxes.at(2)), "");
237  PURIFY_MACRO("NAXIS4", static_cast<t_int>(naxes.at(3)), "");
238  PURIFY_MACRO("CRPIX1", static_cast<t_int>(std::floor(naxes.at(0) / 2)) + 1, "");
239  PURIFY_MACRO("CRPIX2", static_cast<t_int>(std::floor(naxes.at(1) / 2)) + 1, "");
240  PURIFY_MACRO("CRPIX3", 1, "");
241  PURIFY_MACRO("CRPIX4", 1, "");
242  PURIFY_MACRO("CRVAL1", static_cast<t_real>(header.ra * 180. / purify::constant::pi), "");
243  PURIFY_MACRO("CRVAL2", static_cast<t_real>(header.dec * 180. / purify::constant::pi), "");
244  PURIFY_MACRO("CRVAL3", static_cast<t_real>(header.mean_frequency * std::pow(10, 6) * 1.), "");
245  PURIFY_MACRO("CRVAL4", static_cast<t_real>(1), "");
246  PURIFY_MACRO("CTYPE1", "RA---SIN", "");
247  PURIFY_MACRO("CTYPE2", "DEC---SIN", "");
248  PURIFY_MACRO("CTYPE3", "FREQ-OBS", "");
249  PURIFY_MACRO("CTYPE4", "STOKES", "");
250  PURIFY_MACRO("CDELT1", static_cast<t_real>(-header.cell_x / 3600.), "");
251  PURIFY_MACRO("CDELT2", static_cast<t_real>(header.cell_y / 3600.), "");
252  PURIFY_MACRO("CDELT3", static_cast<t_real>(header.channel_width * std::pow(10, 6) * 1.), "");
253  PURIFY_MACRO("CDELT4", 1., "");
254  PURIFY_MACRO("BTYPE", "intensity", "");
255  PURIFY_MACRO("EQUINOX", 2000, "");
256 
257 #undef PURIFY_MACRO
258  write_history(fptr, "SOFTWARE", "Purify", &status);
259  write_history(fptr, "PURIFY-NITERS", header.niters, &status);
260  if (header.hasconverged) {
261  write_history(fptr, "PURIFY-CONVERGED", "True", &status);
262  } else {
263  write_history(fptr, "PURIFY-CONVERGED", "False", &status);
264  }
265  write_history(fptr, "PURIFY-RELATIVEVARIATION", header.relative_variation, &status);
266  write_history(fptr, "PURIFY-RESIDUALCONVERGENCE", header.residual_convergence, &status);
267  write_history(fptr, "PURIFY-EPSILON", header.epsilon, &status);
268  if (fits_write_date(fptr, &status))
269  throw std::runtime_error("Problem writing date in fits file:" + header.fits_name);
270 
271  if (fits_close_file(fptr, &status))
272  throw std::runtime_error("Problem closing fits file:" + header.fits_name);

280  t_int &channels, t_int &pols) {

282  Reads in a cube from a fits file and returns the vector of images.
283  fits_name:: name of fits file
284  */
285 
286  fitsfile *fptr;
287  int status = 0;
288  PURIFY_LOW_LOG("Reading fits file {}", fits_name);
289  if (fits_open_image(&fptr, fits_name.c_str(), READONLY, &status))
290  throw std::runtime_error("Error opening image " + fits_name);
291  const t_int naxis = read_key<int>(fptr, "NAXIS", &status);
292  if (naxis < 1) throw std::runtime_error("Image contains zero axes.");
293  rows = read_key<int>(fptr, "NAXIS1", &status);
294  cols = (naxis > 1) ? read_key<int>(fptr, "NAXIS2", &status) : 1;
295  channels = (naxis > 2) ? read_key<int>(fptr, "NAXIS3", &status) : 1;
296  pols = (naxis > 3) ? read_key<int>(fptr, "NAXIS4", &status) : 1;
297  PURIFY_LOW_LOG("Axes {}", naxis);
298  std::vector<long> fpixel(naxis, 1);
299  PURIFY_LOW_LOG("Dimensions {}x{}x{}x{}", rows, cols, channels, pols);
300  if (pols > 1) throw std::runtime_error("Too many polarisations when reading " + fits_name);
301  t_real nulval = 0;
302  t_int anynul = 0;
303  output.derived() = Vector<typename T::Scalar>::Zero(rows * cols * channels * pols);
304  if (fits_read_pix(fptr, TDOUBLE, fpixel.data(), static_cast<long>(output.size()), &nulval,
305  output.derived().data(), &anynul, &status))
306  throw std::runtime_error("Error reading data from image " + fits_name);
307  if (anynul) PURIFY_LOW_LOG("There are bad values when reading " + fits_name);
308  if (fits_close_file(fptr, &status))
309  throw std::runtime_error("Problem closing fits file: " + fits_name);
310 }
