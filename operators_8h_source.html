135 template <class T>
136 std::tuple<sopt::OperatorFunction<T>, sopt::OperatorFunction<T>> init_zero_padding_2d(
137  const Image<typename T::Scalar> &S, const t_real &oversample_ratio) {
138  const t_uint imsizex_ = S.cols();
139  const t_uint imsizey_ = S.rows();
140  const t_uint ftsizeu_ = std::floor(imsizex_ * oversample_ratio);
141  const t_uint ftsizev_ = std::floor(imsizey_ * oversample_ratio);
142  const t_uint x_start = std::floor(ftsizeu_ * 0.5 - imsizex_ * 0.5);
143  const t_uint y_start = std::floor(ftsizev_ * 0.5 - imsizey_ * 0.5);
144  auto direct = [=](T &output, const T &x) {
145  assert(x.size() == imsizex_ * imsizey_);
146  output = Vector<t_complex>::Zero(ftsizeu_ * ftsizev_);
147 #pragma omp parallel for collapse(2)
148  for (t_uint j = 0; j < imsizey_; j++) {
149  for (t_uint i = 0; i < imsizex_; i++) {
150  const t_uint input_index = utilities::sub2ind(j, i, imsizey_, imsizex_);
151  const t_uint output_index =
152  utilities::sub2ind(y_start + j, x_start + i, ftsizev_, ftsizeu_);
153  output(output_index) = S(j, i) * x(input_index);
154  }
155  }
156  };
157  auto indirect = [=](T &output, const T &x) {
158  assert(x.size() == ftsizeu_ * ftsizev_);
159  output = T::Zero(imsizey_ * imsizex_);
160 #pragma omp parallel for collapse(2)
161  for (t_uint j = 0; j < imsizey_; j++) {
162  for (t_uint i = 0; i < imsizex_; i++) {
163  const t_uint output_index = utilities::sub2ind(j, i, imsizey_, imsizex_);
164  const t_uint input_index = utilities::sub2ind(y_start + j, x_start + i, ftsizev_, ftsizeu_);
165  output(output_index) = std::conj(S(j, i)) * x(input_index);
166  }
167  }
168  };
169  return std::make_tuple(direct, indirect);
170 }

197 std::tuple<sopt::OperatorFunction<T>, sopt::OperatorFunction<T>> init_FFT_2d(
198  const t_uint &imsizey_, const t_uint &imsizex_, const t_real &oversample_factor_,
199  const fftw_plan fftw_plan_flag_ = fftw_plan::measure) {
200  t_int const ftsizeu_ = std::floor(imsizex_ * oversample_factor_);
201  t_int const ftsizev_ = std::floor(imsizey_ * oversample_factor_);
202  t_int plan_flag = (FFTW_MEASURE | FFTW_PRESERVE_INPUT);
203  switch (fftw_plan_flag_) {
204  case (fftw_plan::measure):
205  plan_flag = (FFTW_MEASURE | FFTW_PRESERVE_INPUT);
206  break;
207  case (fftw_plan::estimate):
208  plan_flag = (FFTW_ESTIMATE | FFTW_PRESERVE_INPUT);
209  break;
210  }
211 
212 #ifdef PURIFY_OPENMP_FFTW
213  PURIFY_LOW_LOG("Using OpenMP threading with FFTW.");
214  fftw_init_threads();
215 #endif
216  Vector<typename T::Scalar> src = Vector<t_complex>::Zero(ftsizev_ * ftsizeu_);
217  Vector<typename T::Scalar> dst = Vector<t_complex>::Zero(ftsizev_ * ftsizeu_);
218  // creating plans
219  const auto del = [](fftw_plan_s *plan) { fftw_destroy_plan(plan); };
220  // fftw plan with threads needs to be used before each fftw_plan is created
221 #ifdef PURIFY_OPENMP_FFTW
222  fftw_plan_with_nthreads(omp_get_max_threads());
223 #endif
224  const std::shared_ptr<fftw_plan_s> m_plan_forward(
225  fftw_plan_dft_2d(ftsizev_, ftsizeu_, reinterpret_cast<fftw_complex *>(src.data()),
226  reinterpret_cast<fftw_complex *>(dst.data()), FFTW_FORWARD, plan_flag),
227  del);
228  // fftw plan with threads needs to be used before each fftw_plan is created
229 #ifdef PURIFY_OPENMP_FFTW
230  fftw_plan_with_nthreads(omp_get_max_threads());
231 #endif
232  const std::shared_ptr<fftw_plan_s> m_plan_inverse(
233  fftw_plan_dft_2d(ftsizev_, ftsizeu_, reinterpret_cast<fftw_complex *>(src.data()),
234  reinterpret_cast<fftw_complex *>(dst.data()), FFTW_BACKWARD, plan_flag),
235  del);
236  auto const direct = [m_plan_forward, ftsizeu_, ftsizev_](T &output, const T &input) {
237  assert(input.size() == ftsizev_ * ftsizeu_);
238  output = Matrix<typename T::Scalar>::Zero(input.rows(), input.cols());
239  fftw_execute_dft(
240  m_plan_forward.get(),
241  const_cast<fftw_complex *>(reinterpret_cast<const fftw_complex *>(input.data())),
242  reinterpret_cast<fftw_complex *>(output.data()));
243  output /= std::sqrt(output.size());
244  };
245  auto const indirect = [m_plan_inverse, ftsizeu_, ftsizev_](T &output, const T &input) {
246  assert(input.size() == ftsizev_ * ftsizeu_);
247  output = Matrix<typename T::Scalar>::Zero(input.rows(), input.cols());
248  fftw_execute_dft(
249  m_plan_inverse.get(),
250  const_cast<fftw_complex *>(reinterpret_cast<const fftw_complex *>(input.data())),
251  reinterpret_cast<fftw_complex *>(output.data()));
252  output /= std::sqrt(output.size());
253  };
254  return std::make_tuple(direct, indirect);
255 }
