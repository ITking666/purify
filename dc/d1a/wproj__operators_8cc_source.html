23  PURIFY_HIGH_LOG("du x du: {}, {}", du, dv);
24  if (std::abs(du - dv) > 1e-6)
25  throw std::runtime_error(
26  "Field of view along l and m is not the same, this assumption is required for "
27  "w-projection.");
28  const t_uint rows = u.size();
29  const t_uint cols = ftsizeu_ * ftsizev_;
30  if (u.size() != v.size())
31  throw std::runtime_error(
32  "Size of u and v vectors are not the same for creating gridding matrix.");
33  if (u.size() != w.size())
34  throw std::runtime_error(

36  if (u.size() != weights.size())
37  throw std::runtime_error(

40  throw std::runtime_error(
41  "w kernel size must be at least the size of w=0 kernel, must have Ju <= Jw.");
42  // count gridding coefficients with variable support size
43  Vector<t_int> total_coeffs = Vector<t_int>::Zero(w.size());
44  for (int i = 0; i < w.size(); i++) {
45  const t_int Ju_max =
46  widefield::w_support(w(i), du, static_cast<t_int>(Ju), static_cast<t_int>(Jw));
47  total_coeffs(i) = Ju_max * Ju_max;
48  }
49  const t_real num_of_coeffs = total_coeffs.array().cast<t_real>().sum();
50  PURIFY_HIGH_LOG("Using {} rows (coefficients per a row {}), and memory of {} MegaBytes",
51  total_coeffs.size(), total_coeffs.array().cast<t_real>().mean(),
52  16. * num_of_coeffs / std::pow(10., 6));
53  Sparse<t_complex> interpolation_matrix(rows, cols);
54  try {
55  interpolation_matrix.reserve(total_coeffs);
56  } catch (std::bad_alloc e) {
57  throw std::runtime_error(
58  "Not enough memory for coefficients, choose upper limit on support size Jw.");
59  }

63  const t_uint max_evaluations = 1e8;
64  const t_real relative_error = rel_error;
65  const t_real absolute_error = abs_error;
66  // normalising kernel
67  t_uint e = 0;

72  (dde == dde_type::wkernel_radial) ? 1 : 0.5);

79 #pragma omp parallel for
80  for (t_int m = 0; m < rows; ++m) {
81  // w_projection convolution setup
82  const t_int Ju_max = widefield::w_support(w(m), du, Ju, Jw);
83  t_uint evaluations = 0;

89  for (t_int jv = 1; jv < Ju_max + 1; ++jv) {

91  const t_uint p = utilities::mod(kwv + jv, ftsizev_);
92  const t_uint index = utilities::sub2ind(p, q, ftsizev_, ftsizeu_);
93  interpolation_matrix.insert(m, index) =
94  std::exp(-2 * constant::pi * I * ((kwu + ju) * 0.5 + (kwv + jv) * 0.5)) * weights(m) *
95  ((dde == dde_type::wkernel_radial)
96  ? projection_kernels::exact_w_projection_integration_1d(
97  (u(m) - (kwu + ju)), (v(m) - (kwv + jv)), w_val, du, oversample_ratio,
98  ftkernel_radial, max_evaluations, absolute_error, relative_error,
99  integration::method::p, evaluations)
100  : projection_kernels::exact_w_projection_integration(
101  (u(m) - (kwu + ju)), (v(m) - (kwv + jv)), w_val, du, dv, oversample_ratio,
102  ftkernel_radial, ftkernel_radial, max_evaluations, absolute_error,
103  relative_error, integration::method::h, evaluations));
104 #pragma omp critical(add_eval)
105  {
106  coeffs_done++;
107  if (num_of_coeffs > 100) {
108  if ((coeffs_done % (static_cast<t_int>(num_of_coeffs / 100)) == 0)) {
109  PURIFY_LOW_LOG(
110  "w = {}, support = {}x{}, coeffs: {} of {}, {}%", w_val, Ju_max, Ju_max,
111  coeffs_done, num_of_coeffs,
112  static_cast<t_real>(coeffs_done) / static_cast<t_real>(num_of_coeffs) * 100.);
113  }
114  }
115  }
116  }
117  }
118  }

132 
133  Image<t_complex> gridding_correction = Image<t_complex>::Zero(imsizey_, imsizex_);
134  for (int i = 0; i < gridding_correction.rows(); i++)
135  for (int j = 0; j < gridding_correction.cols(); j++)
136  gridding_correction(i, j) =
137  1. / ftkerneluv(std::sqrt(std::pow((y_start + i + 0.5) / ftsizev_ - 0.5, 2) +
138  std::pow((j + 0.5 + x_start) / ftsizeu_ - 0.5, 2)));
139 
140  return gridding_correction.array() *
141  widefield::generate_chirp(w_mean, cellx, celly, imsizex_, imsizey_).array() * imsizex_ *
